**3. Create packages/modules/auth/src/middleware.ts -- auth macro + requireRole:**
Create two exports:

a) `betterAuthPlugin` -- Elysia plugin with `.macro()` for session injection per RESEARCH.md Pattern 1:
```typescript
import { Elysia } from "elysia";
import { auth } from "./auth";

export const betterAuthPlugin = new Elysia({ name: "better-auth" })
  .macro({
    auth: {
      async resolve({ status, request: { headers } }) {
        const session = await auth.api.getSession({ headers });
        if (!session) return status(401);
        return {
          user: session.user,
          session: session.session,
        };
      },
    },
  });
```

b) `requireRole` function per D-12:
```typescript
export function requireRole(...roles: string[]) {
  return new Elysia({ name: `require-role-${roles.join(",")}` })
    .derive({ as: "scoped" }, async (ctx: any) => {
      const session = await auth.api.getSession({
        headers: ctx.request.headers,
      });
      if (!session) {
        throw new Error("Unauthorized");
      }
      const activeMember = session.session.activeOrganizationId
        ? await auth.api.getFullOrganization({
            headers: ctx.request.headers,
            query: { organizationId: session.session.activeOrganizationId },
          })
        : null;
      // Find user's role in active org
      const member = activeMember?.members?.find(
        (m: any) => m.userId === session.user.id
      );
      if (!member || !roles.includes(member.role)) {
        throw new Error("Forbidden");
      }
      return { memberRole: member.role };
    });
}
```

Note: The exact API for checking role may differ. If `auth.api.getFullOrganization` is not available or has a different signature, use whatever better-auth organization plugin API provides for checking the user's role. The key contract is: given a session with `activeOrganizationId`, resolve the user's role and reject if not in the allowed `roles` array.
