  import { defaultLocale, locales, type Locale } from "@baseworks/i18n";

  /**
   * Parse the NEXT_LOCALE cookie from a raw Cookie header value.
   * Returns null if the cookie is absent or holds an unsupported locale.
   *
   * Phase 19 D-12 — called from the Bun.serve fetch wrapper in apps/api/src/index.ts
   * (Plan 06) once per request, before the Elysia pipeline runs. Replaces the
   * Phase 12 localeMiddleware Elysia plugin that was removed in Plan 01.
   */
  export function parseNextLocaleCookie(cookieHeader: string | null): Locale | null {
    if (!cookieHeader) return null;
    const match = cookieHeader.match(/(?:^|;\s*)NEXT_LOCALE=([^;]+)/);
    if (!match) return null;
    const value = decodeURIComponent(match[1]);
    return (locales as readonly string[]).includes(value)
      ? (value as Locale)
      : null;
  }

**Step 2 — Create `apps/api/src/lib/inbound-trace.ts`** copying PATTERNS.md §"apps/api/src/lib/inbound-trace.ts" lines 720-776 VERBATIM:

  ```typescript
  import ipaddr from "ipaddr.js";
  import { env } from "@baseworks/config";

  // Module-init: parse CIDR list once. Plan 02's validateObservabilityEnv crash-hards
  // on malformed syntax before this module loads, so parseCIDR is safe here.
  const TRUSTED_CIDRS: Array<[ipaddr.IPv4 | ipaddr.IPv6, number]> =
    (env.OBS_TRUST_TRACEPARENT_FROM ?? "")
      .split(",")
      .map((s) => s.trim())
      .filter(Boolean)
      .map((cidr) => ipaddr.parseCIDR(cidr));

  const TRUSTED_HEADER = env.OBS_TRUST_TRACEPARENT_HEADER ?? null;

  /**
   * Decide whether to adopt the inbound W3C traceparent as the parent span,
   * or fall through to a fresh server-side traceId + spanId.
   *
   * Phase 19 D-07 (default never-trust), D-08 (CIDR/header opt-in).
   *
   * @param req - Incoming Request (reads `traceparent` header + optional trusted header)
   * @param remoteAddr - Remote peer IP from `server.requestIP(req)?.address` in Bun.serve
   * @returns traceId + spanId (adopted if trusted, fresh otherwise) + inboundCarrier
   *          (preserves the original traceparent for span.links when untrusted; empty when adopted)
   */
  export function decideInboundTrace(
    req: Request,
    remoteAddr: string,
  ): {
    traceId: string;
    spanId: string;
    inboundCarrier: Record<string, string>;
  } {
    const inbound = req.headers.get("traceparent") ?? "";
    const inboundCarrier: Record<string, string> = inbound
      ? { traceparent: inbound }
      : {};

    let trusted = false;
    if (TRUSTED_CIDRS.length > 0 && remoteAddr) {
      try {
        const addr = ipaddr.parse(remoteAddr);
        trusted = TRUSTED_CIDRS.some(
          ([range, bits]) =>
            addr.kind() === range.kind() && addr.match(range, bits),
        );
      } catch {
        // malformed remote address — treat as untrusted
      }
    }
    if (!trusted && TRUSTED_HEADER && req.headers.get(TRUSTED_HEADER)) {
      trusted = true;
    }

    if (trusted && inbound) {
      // Parse W3C traceparent: 00-<32hex>-<16hex>-<2hex>
      const m = /^00-([0-9a-f]{32})-([0-9a-f]{16})-[0-9a-f]{2}$/.exec(inbound);
      if (m) {
        return { traceId: m[1], spanId: m[2], inboundCarrier: {} };
      }
    }

    // Fresh trace server-side — default path for untrusted or bad-format inbound.
    const traceId = crypto.randomUUID().replace(/-/g, "").slice(0, 32);
    const spanId = crypto.randomUUID().replace(/-/g, "").slice(0, 16);
    return { traceId, spanId, inboundCarrier };
  }
  ```
