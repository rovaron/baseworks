  ```typescript
  import { describe, test, expect, beforeAll } from "bun:test";

  const N = 100;

  // Tenant-toggle: request has x-test-tenant header → we use that as the tenant.
  // Eliminates the real auth.api.getSession dependency; tenantMiddleware is
  // NOT mounted here — we simulate its effect via setTenantContext inline.
  function buildProbeApp(captured: any[]): Elysia {
    const stream = { write: (chunk: string) => { captured.push(JSON.parse(chunk)); } };
    const testLogger = pino(
      { level: "info", mixin: () => obsContext.getStore() ?? {} },
      stream as any,
    );

      .get("/probe", async ({ request }) => {

        setTenantContext({ tenantId: tenantHeader, userId: `U-${tenantHeader}` });
        // Emit a log line that must carry tenantId via the mixin.
        testLogger.info({ at: "probe" }, "probe handler");
        return { tenantId: obsContext.getStore()?.tenantId ?? null };
      });
  }

  async function handle(app: Elysia, tenant: string): Promise<{ body: any; requestId: string }> {

    const res = await obsContext.run(seed, () => app.handle(req));
  ```

**File 2 — `apps/api/__tests__/observability-mixin-perf.test.ts`** (D-28 — W2 — relative-only gate):

  ```typescript
  import { describe, test, expect } from "bun:test";
  import pino from "pino";
  import { obsContext } from "@baseworks/observability";

  function percentile(arr: number[], p: number): number {
    const sorted = [...arr].sort((a, b) => a - b);
    const idx = Math.min(sorted.length - 1, Math.floor(sorted.length * p));
    return sorted[idx];
  }

  function timeLogCalls(logger: any, count: number, inFrame: boolean): number[] {
    const times: number[] = [];
    const work = () => {
      for (let i = 0; i < count; i++) {
        const t0 = performance.now();
        logger.info({ i }, "perf");
        times.push(performance.now() - t0);
      }
    };
    if (inFrame) {
      obsContext.run(
        { requestId: "r", traceId: "t".repeat(32), spanId: "s".repeat(16), locale: "en", tenantId: "T1", userId: "U1" } as any,
        work,
      );
    } else {
      work();
    }
    return times;
  }

  describe("D-28 — pino mixin p99 regression gate (relative ≤5% — W2: no absolute ceiling)", () => {
    test("p99(real mixin) ≤ p99(noop baseline) × 1.05", () => {
      const silentStream = { write: () => {} };
      const baseline = pino({ level: "info", mixin: () => ({}) }, silentStream as any);
      const real = pino({ level: "info", mixin: () => obsContext.getStore() ?? {} }, silentStream as any);

      // Warm-up 1000 calls each (JIT)
      timeLogCalls(baseline, 1000, true);
      timeLogCalls(real, 1000, true);

      const baselineTimes = timeLogCalls(baseline, 10_000, true);
      const realTimes = timeLogCalls(real, 10_000, true);

      const p99Base = percentile(baselineTimes, 0.99);
      const p99Real = percentile(realTimes, 0.99);

      // Sole hard gate (D-28): relative 5% regression budget. W2 — absolute
      // ceiling is INFORMATIONAL only; not asserted (unreliable on Windows CI).
      expect(p99Real).toBeLessThanOrEqual(p99Base * 1.05);

      // Log observed values for tracking + future retrospective analysis.
      // These numbers are INFORMATIONAL — the relative check above is the gate.
      console.log(
        `[D-28 perf gate] p99 baseline=${p99Base.toFixed(4)}ms real=${p99Real.toFixed(4)}ms ratio=${(p99Real / p99Base).toFixed(3)} (informational; not asserted per W2)`,
      );
    }, 30_000);
  });
  ```
