**File 1 — `apps/api/__tests__/observability-context-bleed.test.ts`** (D-27):

  ```typescript
  import { describe, test, expect, beforeAll } from "bun:test";
  import { Elysia } from "elysia";
  import pino from "pino";
  import { obsContext, setTenantContext, setTracer, resetTracer } from "@baseworks/observability";

  const N = 100;

  // Tenant-toggle: request has x-test-tenant header → we use that as the tenant.
  // Eliminates the real auth.api.getSession dependency; tenantMiddleware is
  // NOT mounted here — we simulate its effect via setTenantContext inline.
  function buildProbeApp(captured: any[]): Elysia {
    const stream = { write: (chunk: string) => { captured.push(JSON.parse(chunk)); } };
    const testLogger = pino(
      { level: "info", mixin: () => obsContext.getStore() ?? {} },
      stream as any,
    );
    return new Elysia()
      .get("/probe", async ({ request }) => {
        const tenantHeader = request.headers.get("x-test-tenant") ?? "NONE";
        setTenantContext({ tenantId: tenantHeader, userId: `U-${tenantHeader}` });
        // Emit a log line that must carry tenantId via the mixin.
        testLogger.info({ at: "probe" }, "probe handler");
        return { tenantId: obsContext.getStore()?.tenantId ?? null };
      });
  }

  async function handle(app: Elysia, tenant: string): Promise<{ body: any; requestId: string }> {
    const requestId = `r-${tenant}-${Math.random().toString(36).slice(2, 10)}`;
    const req = new Request("http://x/probe", { headers: { "x-test-tenant": tenant, "x-request-id": requestId } });
    const seed = { requestId, traceId: "t".repeat(32), spanId: "s".repeat(16), locale: "en" as const, tenantId: null, userId: null };
    const res = await obsContext.run(seed, () => app.handle(req));
    return { body: await res.json(), requestId };
  }

  describe("CTX-01 / Success Criterion 5 — 100-RPS concurrent-tenant context bleed gate (D-27)", () => {
    test("100 concurrent interleaved tenantA/tenantB requests — every response + log line carries correct tenantId", async () => {
      const captured: any[] = [];
      const app = buildProbeApp(captured);
      const jobs = Array.from({ length: N }, (_, i) => (i % 2 === 0 ? "A" : "B"));
      const results = await Promise.all(jobs.map(t => handle(app, t)));
      // Every response body reflects its own tenant
      for (let i = 0; i < N; i++) {
        expect(results[i].body.tenantId).toBe(jobs[i]);
      }
      // Every captured log line reflects its own tenant (by matching requestId → expected tenant)
      const requestIdToTenant = new Map(results.map((r, i) => [r.requestId, jobs[i]]));
      const probeLines = captured.filter(l => l.at === "probe");
      expect(probeLines.length).toBe(N);
      for (const line of probeLines) {
        expect(line.tenantId).toBe(requestIdToTenant.get(line.requestId));
      }
    }, 30_000);

    test("100 sequential interleaved requests — same invariant", async () => {
      const captured: any[] = [];
      const app = buildProbeApp(captured);
      const jobs = Array.from({ length: N }, (_, i) => (i % 2 === 0 ? "A" : "B"));
      const results: any[] = [];
      for (const t of jobs) results.push(await handle(app, t));
      for (let i = 0; i < N; i++) {
        expect(results[i].body.tenantId).toBe(jobs[i]);
      }
    }, 30_000);

    test("completion time — 100 concurrent requests finish in under 30 seconds", async () => {
      const captured: any[] = [];
      const app = buildProbeApp(captured);
      const start = performance.now();
      await Promise.all(Array.from({ length: N }, (_, i) => handle(app, i % 2 === 0 ? "A" : "B")));
      const dur = performance.now() - start;
      expect(dur).toBeLessThan(30_000);
    }, 30_000);
  });
  ```

  import { describe, test, expect } from "bun:test";

  function percentile(arr: number[], p: number): number {
    const sorted = [...arr].sort((a, b) => a - b);
    const idx = Math.min(sorted.length - 1, Math.floor(sorted.length * p));
    return sorted[idx];
  }

  function timeLogCalls(logger: any, count: number, inFrame: boolean): number[] {
    const times: number[] = [];
    const work = () => {
      for (let i = 0; i < count; i++) {
