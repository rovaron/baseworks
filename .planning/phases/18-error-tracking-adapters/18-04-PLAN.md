    ```typescript
    /**
     * Pino-sink ErrorTracker adapter (ERR-03 / Phase 18 D-07).
     *
     * Default adapter when ERROR_TRACKER is unset or =pino (plan 01 widens
     * the Phase-17 noop default). Writes through a caller-provided pino
     * logger at the appropriate level with full structured scope. Zero
     * external traffic.
     *
     * Design rules:
     * - scrubPii applied inside captureException BEFORE logger.error (D-12
     *   defense-in-depth — ERR-04 conformance gate).
     * - Breadcrumb ring buffer bounded at 10 entries; oldest-first eviction
     *   (Pitfall: unbounded buffer leaks memory in long-running workers).
     * - withScope provides a closure-scoped scope object — NO instance state
     *   for scope fields, otherwise concurrent requests leak tenantId across
     *   each other (Pitfall 4).
     * - flush always resolves true — pino is synchronous from the adapter POV
     *   (the underlying stream's backpressure is a pino concern, not ours).
     */
    import type { Logger } from "pino";
    import { scrubPii } from "../../lib/scrub-pii";
    import type {
      Breadcrumb,
      CaptureScope,
      ErrorTracker,
      ErrorTrackerScope,
    } from "../../ports/error-tracker";
    import type { LogLevel } from "../../ports/types";

    /** Max breadcrumbs held in the ring; oldest-first eviction on overflow. */
    const BREADCRUMB_BUFFER_SIZE = 10;

    /** Map port LogLevel to pino level name. Default "info". */
    function pinoMethod(logger: Logger, level?: LogLevel): (obj: unknown, msg?: string) => void {
      switch (level) {
        case "error": return logger.error.bind(logger);
        case "warn":  return logger.warn.bind(logger);
        case "debug": return logger.debug.bind(logger);
        case "info":
        default:      return logger.info.bind(logger);
      }
    }

    /**
     * ErrorTracker adapter that forwards to a pino logger at ERROR level with
     * structured scope (tenantId, user_id, request_id, command/queryName),
     * ring-buffered breadcrumbs serialized into `extra.breadcrumbs`.
     */
    export class PinoErrorTracker implements ErrorTracker {
      readonly name = "pino";
      private breadcrumbs: Breadcrumb[] = [];

      constructor(private logger: Logger) {}

      captureException(err: unknown, scope?: CaptureScope): void {
        const errPayload = err instanceof Error
          ? { message: err.message, stack: err.stack, name: err.name }
          : { value: String(err) };

        const raw = {
          err: errPayload,
          user: scope?.user,
          tags: scope?.tags,
          extra: scope?.extra,
          tenantId: scope?.tenantId,
          breadcrumbs: [...this.breadcrumbs],
        };

        const scrubbed = scrubPii(raw) ?? raw;
        this.logger.error(scrubbed, "captured exception");
        this.breadcrumbs = [];
      }

      captureMessage(message: string, level?: LogLevel): void {
        pinoMethod(this.logger, level)(message);
      }

      addBreadcrumb(breadcrumb: Breadcrumb): void {
        this.breadcrumbs.push(breadcrumb);
        if (this.breadcrumbs.length > BREADCRUMB_BUFFER_SIZE) {
          // Oldest-first eviction.
          this.breadcrumbs.splice(0, this.breadcrumbs.length - BREADCRUMB_BUFFER_SIZE);
        }
      }

      withScope<T>(fn: (scope: ErrorTrackerScope) => T): T {
        // Closure-scoped scope — no instance state, no cross-call leakage.
        const local: {
          user?: { id?: string; email?: string } | null;
          tags: Record<string, string>;
          extra: Record<string, unknown>;
          tenantId?: string | null;
        } = { tags: {}, extra: {} };

        const scope: ErrorTrackerScope = {
          setUser: (u) => { local.user = u; },
          setTag: (k, v) => { local.tags[k] = v; },
          setExtra: (k, v) => { local.extra[k] = v; },
          setTenant: (t) => { local.tenantId = t; },
        };

        return fn(scope);
      }

      async flush(_timeoutMs?: number): Promise<boolean> {
        return true;
      }
    }
    ```

| Boundary | Description |
|----------|-------------|
| CaptureScope → pino logger | Adapter receives potentially-PII-bearing scope + err and writes to the process's canonical log stream |
| breadcrumb buffer → captured log | Unbounded buffer would leak memory in long-lived worker processes |
| concurrent requests → shared adapter singleton | One tracker instance serves all requests; scope state must not leak across tenants |

| Threat ID | Category | Component | Disposition | Mitigation Plan |

| T-18-21 | Information disclosure | Cross-tenant scope leak via shared instance state (Pitfall 4) | mitigate | `withScope` uses closure-scoped locals, NO instance fields for scope. Task 1 test 8 runs 2 concurrent callbacks with different tenantIds and asserts mutual non-appearance. Acceptance grep confirms no `this.tags`/`this.user`/etc. |
| T-18-22 | Denial of service | Unbounded breadcrumb buffer leaks memory | mitigate | `BREADCRUMB_BUFFER_SIZE = 10` constant; eviction on overflow. Task 1 test 6 adds 15 and asserts 10 remain. |
| T-18-23 | Denial of service | `flush()` blocks process exit indefinitely | accept | flush always resolves true synchronously — pino streams are a separate backpressure concern not owned by this adapter. Task 1 test 11 asserts. |
