**1. `packages/modules/auth/src/commands/create-invitation.ts`:**

This command handles BOTH email and link invitation modes. The `mode` field determines behavior:
- `mode: "email"` -- uses the provided email address, which triggers the sendInvitationEmail callback to enqueue an email
- `mode: "link"` -- generates a placeholder email `link-invite-{nanoid}@internal`, which triggers the sendInvitationEmail callback BUT the callback detects the @internal suffix and returns early (no email sent)

```typescript
import { Type } from "@sinclair/typebox";
import { defineCommand, ok, err } from "@baseworks/shared";
import { nanoid } from "nanoid";
import { auth } from "../auth";

const CreateInvitationInput = Type.Object({
  email: Type.Optional(Type.String({ format: "email" })),
  role: Type.Union([Type.Literal("admin"), Type.Literal("member")]),
  organizationId: Type.String(),
  mode: Type.Union([Type.Literal("email"), Type.Literal("link")]),
});

export const createInvitation = defineCommand(
  CreateInvitationInput,
  async (input, ctx) => {
    try {
      // For link mode, generate a placeholder @internal email.
      // The sendInvitationEmail callback in auth.ts detects @internal
      // and skips email enqueueing. This is the email suppression contract.
      const email = input.mode === "link"
        ? `link-invite-${nanoid(10)}@internal`
        : input.email!;

      const invitation = await auth.api.createInvitation({
        body: {
          email,
          role: input.role,
          organizationId: input.organizationId,
        },
        headers: new Headers(),
      });
      ctx.emit("invitation.created", {
        invitationId: invitation.id,
        organizationId: input.organizationId,
        email,
        mode: input.mode,
      });
      return ok(invitation);
    } catch (error: any) {
      return err(error.message || "Failed to create invitation");
    }
  },
);
```

The existing routes.ts only has `new Elysia({ name: "auth-routes" }).mount(auth.handler)`. We need to add invitation-specific Elysia routes. These cannot go through better-auth's handler -- they wrap our CQRS commands/queries and add role-based access control.

Updated routes.ts structure:

```typescript
import { Elysia, t } from "elysia";
import { auth } from "./auth";
import { betterAuthPlugin, requireRole } from "./middleware";
import { createInvitation } from "./commands/create-invitation";
import { acceptInvitation } from "./commands/accept-invitation";
import { rejectInvitation } from "./commands/reject-invitation";
import { cancelInvitation } from "./commands/cancel-invitation";
import { listInvitations } from "./queries/list-invitations";
import { getInvitation } from "./queries/get-invitation";

/**
 * Auth routes plugin. Mounts better-auth handler + invitation endpoints.
 */
export const authRoutes = new Elysia({ name: "auth-routes" })
  .mount(auth.handler)
  // Public endpoint: get invitation details for accept page (no auth required)
  // Per D-05: invite accept page needs org name, inviter, role without auth
  .get("/api/invitations/:id", async ({ params }) => {
    const result = await getInvitation.handler(
      { invitationId: params.id },
      { userId: "", tenantId: "", db: null as any, emit: () => {} },
    );
    if (!result.ok) {
      return new Response(JSON.stringify({ error: result.error }), { status: 404 });
    }
    return result.data;
  })
  // Protected invitation endpoints (require auth + owner/admin role)
  .group("/api/invitations", (app) =>
    app
      .use(betterAuthPlugin)
      .use(requireRole("owner", "admin"))
      // Create invitation (email or link mode) - per D-04, D-13, D-14, D-15
      // mode field determines email vs link: "email" sends real email, "link" uses @internal placeholder
      .post("/", async ({ body, user, session }: any) => {
        const activeOrgId = session.activeOrganizationId;
        if (!activeOrgId) {
          return new Response(JSON.stringify({ error: "No active organization" }), { status: 400 });
        }
        const result = await createInvitation.handler(
          {
            email: body.mode === "email" ? body.email : undefined,
            role: body.role,
            organizationId: activeOrgId,
            mode: body.mode,
          },
          { userId: user.id, tenantId: activeOrgId, db: null as any, emit: () => {} },
        );
        if (!result.ok) {
          return new Response(JSON.stringify({ error: result.error }), { status: 400 });
        }
        return result.data;
      }, {
        body: t.Object({
          email: t.Optional(t.String({ format: "email" })),
          role: t.Union([t.Literal("admin"), t.Literal("member")]),
          mode: t.Union([t.Literal("email"), t.Literal("link")]),
        }),
        auth: true,
      })
      // List pending invitations for active org - per D-03, INVT-05
      .get("/", async ({ session }: any) => {
        const activeOrgId = session.activeOrganizationId;
        if (!activeOrgId) {
          return new Response(JSON.stringify({ error: "No active organization" }), { status: 400 });
        }
        const result = await listInvitations.handler(
          { organizationId: activeOrgId },
          { userId: "", tenantId: activeOrgId, db: null as any, emit: () => {} },
        );
        if (!result.ok) {
          return new Response(JSON.stringify({ error: result.error }), { status: 400 });
        }
        return result.data;
      }, { auth: true })
      // Cancel invitation - per D-12, INVT-05
      .delete("/:id", async ({ params, session }: any) => {
        const activeOrgId = session.activeOrganizationId;
        if (!activeOrgId) {
          return new Response(JSON.stringify({ error: "No active organization" }), { status: 400 });
        }
        const result = await cancelInvitation.handler(
          { invitationId: params.id, organizationId: activeOrgId },
          { userId: "", tenantId: activeOrgId, db: null as any, emit: () => {} },
        );
        if (!result.ok) {
          return new Response(JSON.stringify({ error: result.error }), { status: 400 });
        }
        return result.data;
      }, { auth: true })
      // Resend invitation - per INVT-05
      .post("/:id/resend", async ({ params, session }: any) => {
        const activeOrgId = session.activeOrganizationId;
        if (!activeOrgId) {
          return new Response(JSON.stringify({ error: "No active organization" }), { status: 400 });
        }
        // Cancel then re-create is one approach, or use resend flag
        // Try auth.api.createInvitation with same email + cancelPendingInvitationsOnReInvite
        // For now: get invitation, then create again with same email/role
        const getResult = await getInvitation.handler(
          { invitationId: params.id },
          { userId: "", tenantId: "", db: null as any, emit: () => {} },
        );
        if (!getResult.ok) {
          return new Response(JSON.stringify({ error: "Invitation not found" }), { status: 404 });
        }
        const inv = getResult.data as any;
        // Re-create invitation with same email/role
        // Detect mode from email: @internal = link mode, else email mode
        const mode = inv.email.endsWith("@internal") ? "link" : "email";
        const result = await createInvitation.handler(
          { email: mode === "email" ? inv.email : undefined, role: inv.role, organizationId: activeOrgId, mode },
          { userId: "", tenantId: activeOrgId, db: null as any, emit: () => {} },
        );
        if (!result.ok) {
          return new Response(JSON.stringify({ error: result.error }), { status: 400 });
        }
        return result.data;
      }, { auth: true }),
  );
```
