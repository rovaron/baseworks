**1. packages/modules/auth/src/commands/create-tenant.ts:**
```typescript
import { Type } from "@sinclair/typebox";
import { defineCommand, ok, err } from "@baseworks/shared";
import { auth } from "../auth";

const CreateTenantInput = Type.Object({
  name: Type.String({ minLength: 1, maxLength: 100 }),
  slug: Type.Optional(Type.String({ minLength: 1, maxLength: 50, pattern: "^[a-z0-9-]+$" })),
});

export const createTenant = defineCommand(CreateTenantInput, async (input, ctx) => {
  try {
    const org = await auth.api.createOrganization({
      body: {
        name: input.name,
        slug: input.slug || input.name.toLowerCase().replace(/[^a-z0-9]+/g, "-").slice(0, 50),
      },
      headers: new Headers(),
    });
    ctx.emit("tenant.created", { tenantId: org.id, createdBy: ctx.userId });
    return ok(org);
  } catch (error: any) {
    return err(error.message || "Failed to create tenant");
  }
});
```

**1. Create packages/modules/auth/src/commands/update-profile.ts:**
Per D-17: User can update name, email, avatar URL, password.
```typescript
import { Type } from "@sinclair/typebox";
import { defineCommand, ok, err } from "@baseworks/shared";
import { auth } from "../auth";

const UpdateProfileInput = Type.Object({
  name: Type.Optional(Type.String({ minLength: 1, maxLength: 100 })),
  email: Type.Optional(Type.String({ format: "email" })),
  image: Type.Optional(Type.String()),
  currentPassword: Type.Optional(Type.String()),
  newPassword: Type.Optional(Type.String({ minLength: 8, maxLength: 128 })),
});

export const updateProfile = defineCommand(UpdateProfileInput, async (input, ctx) => {
  try {
    // Update basic profile fields via better-auth user update API
    const updateData: Record<string, any> = {};
    if (input.name !== undefined) updateData.name = input.name;
    if (input.email !== undefined) updateData.email = input.email;
    if (input.image !== undefined) updateData.image = input.image;

    if (Object.keys(updateData).length > 0) {
      await auth.api.updateUser({
        body: updateData,
        headers: new Headers(),
      });
    }

    // Password change requires separate API call
    if (input.newPassword && input.currentPassword) {
      await auth.api.changePassword({
        body: {
          currentPassword: input.currentPassword,
          newPassword: input.newPassword,
        },
        headers: new Headers(),
      });
    }

    return ok({ updated: true });
  } catch (error: any) {
    return err(error.message || "Failed to update profile");
  }
});
```

Note: The exact better-auth API for `updateUser` and `changePassword` may differ. Adapt method names to match actual API. The key contract: update user fields and optionally change password.

**2. Create packages/modules/auth/src/queries/get-profile.ts:**

IMPORTANT: Do NOT use `auth.api.getSession({ headers: new Headers() })` -- passing empty Headers will always return null since there is no session cookie. Instead, use `ctx.userId` (available from HandlerContext) and query the user table directly via `ctx.db`.

However, auth tables are not tenant-scoped (per Pitfall 6), so `ctx.db` (which is `scopedDb`) may not work for the `user` table. Instead, import the raw db connection and the user schema to query directly:

```typescript
import { Type } from "@sinclair/typebox";
import { defineQuery, ok, err } from "@baseworks/shared";
import { createDb } from "@baseworks/db";
import { user as userTable } from "@baseworks/db/src/schema/auth";
import { eq } from "drizzle-orm";
import { env } from "@baseworks/config";

const GetProfileInput = Type.Object({});

const db = createDb(env.DATABASE_URL);

export const getProfile = defineQuery(GetProfileInput, async (_input, ctx) => {
  if (!ctx.userId) return err("Not authenticated");

  try {
    const users = await db
      .select({
        id: userTable.id,
        name: userTable.name,
        email: userTable.email,
        image: userTable.image,
        emailVerified: userTable.emailVerified,
        createdAt: userTable.createdAt,
      })
      .from(userTable)
      .where(eq(userTable.id, ctx.userId))
      .limit(1);

    if (!users.length) return err("User not found");
    return ok(users[0]);
  } catch (error: any) {
    return err(error.message || "Failed to get profile");
  }
});
```

This avoids the empty-Headers problem entirely and queries the user table directly with the authenticated userId from context. No password hash or sensitive fields are exposed -- only the explicitly selected columns.

If the db import pattern above causes issues (circular dependency with createDb), alternatively reuse the db instance from the auth module:
```typescript
import { auth } from "../auth";
// Use whatever db reference the auth module already has
```

The key requirement: get-profile MUST use `ctx.userId` to identify the user, NOT `auth.api.getSession` with empty headers.
